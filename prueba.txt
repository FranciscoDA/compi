VAR
	[FIB_A, FIB_B] = [INTEGER, INTEGER]
	[CONTADOR] = [INTEGER]
ENDVAR

PRINT "Hola mundo"
CONTADOR := 0
FIB_A := 1
FIB_B := 0
WHILE CONTADOR < 5 DO
	FIB_A := FIB_A + FIB_B
	FIB_B := FIB_A - FIB_B
	CONTADOR := CONTADOR + 1
END WHILE
PRINT "FIB(", CONTADOR, ") = ", FIB_B
PRINT "Fin"
